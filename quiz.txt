from item_categories
left join items on items.id = item_categories.items_id 
left join categories on categories.id = item_categories.categories_id
where categories.name = 'main dish';

Select categories.id as '', categories.name as '', group_concat(items.name) as '', max(items.price)as ''
from item_categories
left join items on items.id = item_categories.items_id 
left join categories on categories.id = item_categories.categories_id
group by categories.id
order by 1
